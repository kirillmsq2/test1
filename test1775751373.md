@startuml
title Оплата QR‑кода через сервис ERIP

actor "Клиент (моб. приложение)" as Client
participant "Точка продажи (POS)" as POS
participant "Система ERIP (платёжный сервис)" as ERIP
participant "Банк‑эмитент (плательщик)" as BankPayer
participant "Банк‑бенефициар" as BankBenef
participant "Система антифрод" as AntiFraud
participant "Система СМП (межбанковский процессинг)" as SMP
database "Директория RTP" as RTP

== Формирование QR‑кода ==
POS -> ERIP : CreateEnvoy(корзина)
ERIP -> RTP : StoreEnvoy()
RTP --> ERIP : EnvoyID
ERIP --> POS : EnvoyID

== Платёж клиента ==
Client -> ERIP : PayEnvoy(EnvoyID)
ERIP -> AntiFraud : GetRiskIndicator(Client)
AntiFraud --> ERIP : RiskIndicator

alt Межбанковский платёж
    ERIP -> BankPayer : RequestFunds(EnvoyID, RiskIndicator)
    BankPayer --> ERIP : FundsReserved
    ERIP -> SMP : TransferFunds(Payer→Beneficiary)
    SMP --> ERIP : TransferConfirmed
else Внутрибанковский платёж
    ERIP -> BankBenef : DebitAccount(Payer, Amount)
    BankBenef --> ERIP : DebitConfirmed
end

ERIP -> POS : PaymentConfirmed(EnvoyID)
POS --> Client : PaymentResult(success)

@enduml
